145inline array<double, 4> calc(double s_q, double s_v, double t_q, double t_v) {
146 return {s_q, s_v, 3 * (t_q - s_q) - 2 * s_v - 1 * t_v, -2 * (t_q - s_q) + s_v + t_v};
147}

149template <std::size_t DOF>
150inline array<array<double, 4>, DOF> calc_plan(const Joints<DOF>& current_q, const Joints<DOF>& current_v,
151 const Joints<DOF>& target_q, const Joints<DOF>& target_v) {
152 array<array<double, 4>, DOF> plan_params;
153 for (int i = 0; i < plan_params.size(); i++)
154 plan_params[i] = calc(current_q[i], current_v[i] * 1, target_q[i], target_v[i]);
155 return plan_params;
156}
157
158template <std::size_t DOF>
159inline Joints<DOF> plan_infer(const array<array<double, 4>, DOF>& plan_params, double t) {
160 auto ret = Joints<DOF>();
161 for (int i = 0; i < plan_params.size(); i++)
162 ret[i] = plan_params[i][0] + plan_params[i][1] * t + plan_params[i][2] * t * t + plan_params[i][3] * t * t * t;
163 return ret;
164}

48 inline T max_diff(const std::vector<T>& vec1, const std::vector<T>& vec2) {
49  T mdiff = 0;
50  for (size_t i = 0; i < vec1.size(); i++) {
51  T diff = std::abs(vec1[i] - vec2[i]);
52  if (diff > mdiff) mdiff = diff;
53  }
54  return mdiff;
55 }

57 inline std::vector<double> calc(double s_q, double s_v, double t_q, double t_v) {
58  return (fabs(s_q - t_q) < CALC_ERROR)
59  ? std::vector<double>{t_q, 0, 0, 0}
60  : std::vector<double>{s_q, s_v, 3 * (t_q - s_q) - 2 * s_v - t_v, -2 * (t_q - s_q) + s_v + t_v};
61 }
